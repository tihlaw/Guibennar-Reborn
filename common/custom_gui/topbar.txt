
#raj menubar
custom_window = {
	name = raj_menubar_window
	potential = {
		OR = { 
			overlord = { has_government_attribute = is_raja }
			has_government_attribute = is_raja
			any_known_country = {
				OR = { 
					overlord = { has_government_attribute = is_raja }
					has_government_attribute = is_raja
				}
			}
		}
	}
}

custom_icon = {
	name = raj_menubar_bg 
	potential = { always = yes }
}

custom_text_box = {
	name = raj_menubar_value
	potential = { always = yes }
}

custom_button = {
	name = raj_menubar_button
	tooltip = raj_menubar_button_tt
	potential = { always = yes }
	
	trigger = {
		always = yes 
	}
	
	effect = {
		if = {
			limit = { NOT = { has_country_flag = raj_menu_open } }
			set_country_flag = raj_menu_open
			custom_tooltip = raj_open_raj_menu_tt
			hidden_effect = { country_event = { id = rahenraj.3000 } }
		}
	}
}
